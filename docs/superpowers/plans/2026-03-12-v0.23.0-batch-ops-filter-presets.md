```python
    """Queue a single wanted item for re-translation. Returns job_id or None if not found."""
```

```python
@wanted_bp.route("/batch-translate", methods=["POST"])
def batch_translate():
    """Queue multiple wanted items for re-translation.

    Request body: {"item_ids": [1, 2, 3]}
    Returns: {"queued": N, "job_ids": [...]}
    """
    data = request.get_json(force=True, silent=True) or {}
    item_ids = data.get("item_ids")
    if not item_ids:
        return jsonify({"error": "item_ids required and must be non-empty"}), 400

    job_ids = []
    for item_id in item_ids:
        job_id = _retranslate_item(item_id)
        if job_id:
            job_ids.append(job_id)

    return jsonify({"queued": len(job_ids), "job_ids": job_ids}), 202
```
