```python
def test_translation_failure_cleans_up_temp_file(self, temp_db, monkeypatch, tmp_path):
    import translator

    mkv = tmp_path / "episode.mkv"

    # Simulate an embedded ASS stream being detected

        out = translator.get_output_path_for_lang(str(mkv))
        from pathlib import Path
        Path(out).write_text("partial content")
        raise RuntimeError("translation crashed mid-write")

    monkeypatch.setattr(translator, "_translate_srt_from_stream_impl", crashing_translate, raising=False)
    monkeypatch.setattr(translator, "translate_srt_from_stream", crashing_translate, raising=False)

    with pytest.raises(RuntimeError):
        translator.translate_file(str(mkv))

    output_path = translator.get_output_path_for_lang(str(mkv))
    from pathlib import Path
    assert not Path(output_path).exists(), f"Temp file was not cleaned up: {output_path}"
```

```python
class TestComputeRetryAfter:
    """_compute_retry_after: exponential backoff with configurable cap."""

    def test_first_search_returns_approximately_base_delay(self):
        from wanted_search import _compute_retry_after
        settings = _make_settings(wanted_backoff_base_hours=2, wanted_backoff_cap_hours=48)
        result = _compute_retry_after(search_count=1, settings=settings)
        assert result is not None
        ts = datetime.fromisoformat(result)
        now = datetime.now(timezone.utc)
        delta_hours = (ts - now).total_seconds() / 3600
        assert 1.9 <= delta_hours <= 2.1, f"Expected ~2h delay, got {delta_hours:.2f}h"

    def test_each_retry_increases_delay(self):
        from wanted_search import _compute_retry_after
        settings = _make_settings(wanted_backoff_base_hours=1, wanted_backoff_cap_hours=48)
        now = datetime.now(timezone.utc)

        delays = []
        for count in range(1, 5):
            ts = datetime.fromisoformat(_compute_retry_after(count, settings))
            delays.append((ts - now).total_seconds() / 3600)

        for i in range(len(delays) - 1):
            assert delays[i + 1] > delays[i], f"Delay did not increase at step {i + 1}"

    def test_cap_is_respected(self):
        from wanted_search import _compute_retry_after
        settings = _make_settings(wanted_backoff_base_hours=1, wanted_backoff_cap_hours=10)
        result = _compute_retry_after(search_count=20, settings=settings)
        ts = datetime.fromisoformat(result)
        now = datetime.now(timezone.utc)
        delay_hours = (ts - now).total_seconds() / 3600
        assert delay_hours <= 10.5, f"Cap exceeded: {delay_hours:.2f}h (cap=10)"

    def test_adaptive_backoff_disabled_returns_none(self):
        """When adaptive_backoff_enabled=False, function should return None."""
        from wanted_search import _compute_retry_after
        settings = _make_settings(wanted_adaptive_backoff_enabled=False)
        result = _compute_retry_after(search_count=3, settings=settings)
        assert result is None, "Expected None when adaptive backoff is disabled"
```

- [ ] **Step 4.2: Write cache tests**

Note on `search()` signature: `ProviderManager.search()` has NO `cache_ttl` parameter.
The TTL is read from `settings.provider_cache_ttl_minutes` internally. To control cache
behaviour in tests, monkeypatch that setting AND mock out the cache lookup so the test
is deterministic regardless of Redis/DB availability.

```python
class TestProviderManagerCache:
    """search() must cache results and skip provider on cache hit."""

    def test_cache_hit_skips_provider_on_second_search(self, app_ctx, monkeypatch):
        from providers import ProviderManager

        manager = ProviderManager()
        provider, result = _make_mock_provider()

        # Register exactly one provider and clear any from _init_providers
        manager._providers.clear()
        manager._providers["test_provider"] = provider
        manager._circuit_breakers["test_provider"] = MagicMock(is_open=False)

        # Enable caching via settings (TTL > 0)
        monkeypatch.setattr(manager, "_get_cache_backend", lambda: MagicMock(
            get=MagicMock(return_value=None),   # first call: cache miss
            set=MagicMock(),
        ))

        # After first call, cache backend returns the stored results
        stored = []

        class ControlledCache:
            def get(self, key):
                return stored[0] if stored else None
            def set(self, key, value, ex=None):
                stored.append(value)

        monkeypatch.setattr(manager, "_get_cache_backend", lambda: ControlledCache())

        query = MagicMock()
        query.series_title = "CacheTest"
        query.season = 1
        query.episode = 1
        query.target_language = "de"
```
