> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Extend the existing glossary system with auto-suggestion (frequency analysis + LLM NER over subtitle files), per-entry metadata (type, confidence, approved flag), TSV export, and a `SUBLARR_GLOSSARY_ENABLED` setting — so users can build consistent term glossaries with minimal manual effort.

**Architecture:** The existing `glossary_entries` table, CRUD routes, frontend panel, and translation injection (in `translator.py` / `llm_utils.py`) are already in place. This plan adds: (1) a DB migration for three new columns, (2) a `glossary_extractor.py` service for candidate detection, (3) two new API endpoints (suggest + TSV export), (4) frontend suggest-and-approve UI, and (5) minor config/settings additions.

**Tech Stack:** Flask + SQLAlchemy + Alembic, pysubs2 (subtitle parsing), collections.Counter (frequency), existing Ollama LLM backend, React 19 + TypeScript + TanStack Query, Tailwind v4 CSS vars.

```python
def build_prompt_with_glossary(
    prompt_template: str,
    glossary_entries: list[dict] | None,
    lines: list[str],
) -> str:
    """Build a translation prompt with glossary terms as a structured block.

    Args:
        prompt_template: Base prompt template
        glossary_entries: List of {source_term, target_term, approved?} dicts (max 50)
        lines: Source lines to translate

    Returns:
        Complete prompt with <glossary> block and numbered lines
    """
    if not glossary_entries:
        return build_translation_prompt(prompt_template, lines)

    # Filter to approved only, cap at 50
    approved = [e for e in glossary_entries if e.get("approved", 1)]
    capped = approved[:50]

    glossary_lines = "\n".join(
        f"  {e['source_term']} → {e['target_term']}" for e in capped
    )
    glossary_block = f"<glossary>\n{glossary_lines}\n</glossary>\n\n"

    numbered = "\n".join(f"{i + 1}. {line}" for i, line in enumerate(lines))
    return glossary_block + prompt_template + "\n" + numbered
```

**Files:**

The extractor:
1. Finds all subtitle sidecar files for a series (`.ass`, `.srt`) via filesystem scan
2. Parses them with `pysubs2`, collects all dialogue lines
3. Runs frequency analysis (`Counter`) to find repeated tokens ≥ `min_freq` threshold
4. Applies simple heuristics to classify candidates: title-case words = `character`, place-name patterns = `place`, else `other`
5. Returns `list[dict]` — each with `source_term`, `term_type`, `frequency`, `confidence`
6. **No LLM call** — pure frequency + heuristic. LLM suggestion is deferred to avoid blocking the sync request path.

```python
"""Glossary candidate extraction from subtitle sidecar files.

Uses frequency analysis and simple heuristics — no LLM required.
"""
from __future__ import annotations

import logging
import re
from collections import Counter
from pathlib import Path

import pysubs2

logger = logging.getLogger(__name__)

# Simple heuristics for term classification
_PLACE_SUFFIXES = re.compile(
    r"\b(village|town|city|forest|valley|mountain|island|realm|land|district|gate|clan)\b",
    re.IGNORECASE,
)
_TITLE_WORD = re.compile(r"^[A-Z][a-z]{2,}$")
_ALL_CAPS = re.compile(r"^[A-Z]{2,}$")


def _classify_term(term: str) -> str:
    """Heuristically classify a term as character, place, or other."""
    if _PLACE_SUFFIXES.search(term):
        return "place"
    if _TITLE_WORD.match(term.split()[-1] if " " in term else term):
        return "character"
    return "other"


def _collect_lines_from_file(path: Path) -> list[str]:
    """Parse an ASS or SRT file and return all dialogue text lines."""
    try:
        subs = pysubs2.load(str(path))
        return [event.text for event in subs if event.text.strip()]
    except Exception as exc:
        logger.warning("Failed to parse subtitle file %s: %s", path, exc)
        return []


def extract_candidates(
    subtitle_dir: str,
    source_lang: str,
    min_freq: int = 3,
    max_candidates: int = 100,
) -> list[dict]:
    """Extract glossary term candidates from subtitle files in a directory.

    Scans all .ass and .srt files matching the source language, collects
    dialogue, runs frequency analysis, and returns candidates sorted by
    frequency descending.

    Args:
        subtitle_dir: Directory to scan (usually the series media root)
        source_lang: Two-letter language code (e.g. 'en') to filter files
        min_freq: Minimum occurrence count to include a term
        max_candidates: Maximum number of candidates to return

    Returns:
        List of dicts: {source_term, term_type, frequency, confidence}
    """
    base = Path(subtitle_dir)
    if not base.is_dir():
        return []

    # Collect subtitle files for the source language
    patterns = [f"*.{source_lang}.ass", f"*.{source_lang}.srt"]
    files: list[Path] = []
    for pat in patterns:
        files.extend(base.rglob(pat))

    if not files:
        return []

    # Gather all words across all dialogue lines
    all_lines: list[str] = []
    for f in files:
        all_lines.extend(_collect_lines_from_file(f))

    # Tokenize: title-cased words only (likely proper nouns)
    token_re = re.compile(r"\b([A-Z][a-zA-Z\-']{2,}(?:\s[A-Z][a-zA-Z\-']{2,})?)\b")
    counter: Counter[str] = Counter()
    for line in all_lines:
        # Strip ASS override tags before tokenizing
        clean = re.sub(r"\{[^}]*\}", "", line)
        for match in token_re.finditer(clean):
            counter[match.group(1)] += 1

    candidates = []
    for term, freq in counter.most_common():
        if freq < min_freq:
            break
        term_type = _classify_term(term)
        # Confidence heuristic: higher frequency → higher confidence, capped at 0.99
        confidence = min(0.5 + (freq / 100), 0.99)
        candidates.append(
            {
                "source_term": term,
                "term_type": term_type,
                "frequency": freq,
                "confidence": round(confidence, 3),
            }
        )
        if len(candidates) >= max_candidates:
            break

    return candidates
```

```python
@bp.route("/glossary/export", methods=["GET"])
def export_glossary_tsv():
    """Export glossary as TSV (tab-separated values).

    ---
    tags: [glossary]
    parameters:
      - name: series_id
        in: query
        schema:
          type: integer
        description: Series ID. Omit for global glossary.
    responses:
      200:
        description: TSV file download
        content:
          text/tab-separated-values:
            schema:
              type: string
    """
    from db.translation import get_glossary_entries

    series_id = request.args.get("series_id", type=int)
    entries = get_glossary_entries(series_id)

    lines = ["source_term\ttarget_term\tterm_type\tnotes"]
    for e in entries:
        notes = (e.get("notes") or "").replace("\t", " ")
        lines.append(
            f"{e['source_term']}\t{e['target_term']}\t{e.get('term_type', 'other')}\t{notes}"
        )

    filename = f"glossary_series_{series_id}.tsv" if series_id else "glossary_global.tsv"
    return (
        "\n".join(lines),
        200,
        {
            "Content-Type": "text/tab-separated-values; charset=utf-8",
            "Content-Disposition": f'attachment; filename="{filename}"',
        },
    )
```

```typescript
export function useSuggestGlossaryTerms() {
  return useMutation({
    mutationFn: ({ seriesId, options }: {
      seriesId: number
      options?: { source_lang?: string; min_freq?: number }
    }) => suggestGlossaryTerms(seriesId, options),
  })
}

export function useExportGlossaryTsv() {
  return useMutation({
    mutationFn: ({ seriesId }: { seriesId?: number | null }) => exportGlossaryTsv(seriesId),
    onSuccess: (blob, { seriesId }) => {
      // Trigger browser download
      const url = URL.createObjectURL(blob)
      const a = document.createElement('a')
      a.href = url
      a.download = seriesId != null ? `glossary_series_${seriesId}.tsv` : 'glossary_global.tsv'
      a.click()
      URL.revokeObjectURL(url)
    },
  })
}

export function useApproveGlossaryEntry() {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: ({ entryId, seriesId }: { entryId: number; seriesId?: number | null }) =>
      updateGlossaryEntry(entryId, { approved: 1 }),
    onSuccess: (_, { seriesId }) => {
      queryClient.invalidateQueries({ queryKey: ['glossary', seriesId] })
      queryClient.invalidateQueries({ queryKey: ['glossary', 'global'] })
    },
  })
}
```
